## Install

```bash
npm install --save funnel-react
```

> or

```bash
yarn add funnel-react
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'funnel-react'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
